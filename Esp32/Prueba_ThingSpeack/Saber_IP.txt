import network
import utime

wf = network.WLAN(network.STA_IF)
wf.active(True)
lista = wf.scan()
for red in lista:
    print(red[0].decode()+""+str(red[3]))
    
wf.connect("nomre de la red","calve")
while not wf.isconnected():
    print(".")
    utime.sleep(1)
    
print(wf.ifconfig()[0])
